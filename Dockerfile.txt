# ── Stage 1: Install dependencies ──────────────────────────────────────────
FROM node:22-alpine AS deps
WORKDIR /app

# Copy package files and install production + dev dependencies
COPY package.json package-lock.json ./
RUN npm ci --ignore-scripts

# ── Stage 2: Build the Next.js app ─────────────────────────────────────────
FROM node:22-alpine AS builder
WORKDIR /app

# Copy installed node_modules from deps stage
COPY --from=deps /app/node_modules ./node_modules
COPY . .

# Disable Next.js telemetry during build
ENV NEXT_TELEMETRY_DISABLED=1

# Build the application (includes generating static pages)
RUN npm run build

# ── Stage 3: Production runtime ────────────────────────────────────────────
FROM node:22-alpine AS runner
WORKDIR /app

# Create data directory for SQLite and CSV files
RUN mkdir -p /app/data

# Copy necessary files from builder
COPY --from=builder /app/package.json ./
COPY --from=builder /app/.next ./.next
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/public ./public
COPY --from=builder /app/data ./data

# Set environment variables (defaults; override with -e at runtime)
ENV NODE_ENV=production
ENV PORT=3000
ENV BRIDGE_API_KEYS=test-key-123,prod-key-789

# Next.js telemetry off
ENV NEXT_TELEMETRY_DISABLED=1

# Expose the port the app listens on
EXPOSE 3000

# Start the Next.js server
CMD ["npm", "run", "start"]
